def get_strict_time():
    """Returns exactly formatted time for logging"""
    return datetime.now(IST).strftime("%d-%b-%Y %I:%M:%S %p").upper()

def safe_float(val, default=0.0):
    """Prevents Python from crashing if an old save file has '--' instead of numbers"""
    try: return float(val)
    except: return default

# --- UTILS & LOGGING ---
def write_to_log_file(log_text):
    """Writes terminal logs to a daily text file"""
    try:
        log_dir = os.path.join(BASE_DIR, "logs", datetime.now(IST).strftime("%Y-%m-%d"))
        os.makedirs(log_dir, exist_ok=True)
        with open(os.path.join(log_dir, "app.log"), "a", encoding="utf-8") as f:
            f.write(f"[{get_strict_time()}] {log_text}\n")
    except: pass

def add_log(action, entry="--", sl="--", target="--", expiry="--", exit_price="--", pnl="--", highlight=""):
    """Pushes a new log to the frontend terminal array"""
    time_str = get_strict_time()
    SYSTEM_LOGS.append({"time": time_str, "action": action, "entry": entry, "sl": sl, "target": target, "expiry": expiry, "exit": exit_price, "pnl": pnl, "highlight": highlight})
    if len(SYSTEM_LOGS) > 100: SYSTEM_LOGS.pop(0) # Keep max 100 logs in memory
    write_to_log_file(action)

def load_local_orders():
    """Loads previous Paper Trades from local JSON file to persist data after restart"""
    global TRADE_HISTORY, PAPER_PNL, PAPER_PROFIT, PAPER_LOSS, PAPER_BALANCE
    global IN_POSITION, CURRENT_TRADE, TRADE_ENTRY_PRICE, TRADE_SL, TRADE_TARGET, TRADE_EXPIRY, CURRENT_SIGNAL, CURRENT_SYMBOL
    if os.path.exists(LOCAL_ORDERS_FILE):
        try:
            with open(LOCAL_ORDERS_FILE, "r") as f: TRADE_HISTORY = json.load(f)
            restored_count = len(TRADE_HISTORY)
            for o in TRADE_HISTORY:
                # Patch missing data for backward compatibility
                if "qty" not in o: o["qty"] = "25"
                if "sl" not in o: o["sl"] = "--"
                if "target" not in o: o["target"] = "--"
                if "type" not in o: o["type"] = "BUY CE"

                if o.get("status") == "CLOSED" and o.get("pnl") != "--":
                    val = safe_float(o["pnl"])
                    PAPER_PNL += val; PAPER_BALANCE += val
                    if val > 0: PAPER_PROFIT += val
                    else: PAPER_LOSS += abs(val)
                elif o.get("status") == "OPEN":
                    IN_POSITION = True
                    CURRENT_TRADE = "CE" if "CE" in o.get("type", "") else "PE"
                    CURRENT_SYMBOL = str(o.get("symbol", o.get("type", "").replace("BUY ", "")))
                    TRADE_ENTRY_PRICE = safe_float(o.get("entry", 0))
                    TRADE_EXPIRY = o.get("expiry", "--")
                    TRADE_SL = o.get("sl", "--")
                    TRADE_TARGET = o.get("target", "--")
                    CURRENT_SIGNAL = "HOLDING ACTIVE TRADE(S)"
            add_log(f"💾 Restored {restored_count} trades from memory.")
        except: pass

def save_local_orders():
    """Saves Paper Trades to local JSON file"""

    except: pass

def record_trade_entry(trade_type, entry_price, expiry, qty="25", sl="--", target="--", order_id=None, symbol="--"):
    """Records a new buy entry into the memory and JSON"""
    if order_id == None: order_id = f"SIM-{int(time.time())}" 
    TRADE_HISTORY.append({
        "id": order_id, "time": get_strict_time(), "symbol": symbol,
        "type": trade_type, "entry": round(entry_price, 2), "exit": "--",
        "expiry": expiry, "qty": str(qty), "sl": str(sl), "target": str(target),
        "status": "OPEN", "pnl": "--"
    })
    save_local_orders()

def record_trade_exit(exit_price, pnl):
    """Updates the last open trade with exit price and PnL"""
    if TRADE_HISTORY and TRADE_HISTORY[-1]["status"] == "OPEN":
        TRADE_HISTORY[-1]["exit"] = round(exit_price, 2)
        TRADE_HISTORY[-1]["pnl"] = round(pnl, 2)
        TRADE_HISTORY[-1]["status"] = "CLOSED"
        save_local_orders()

def get_next_expiry():
    """Calculates the upcoming Thursday expiry date"""
    t = datetime.now(IST).date()
    exp = t + timedelta(days=(3 - t.weekday()) % 7)
    return exp.strftime("%d%b%y").upper(), exp.strftime("%d-%b-%Y").upper()

def check_market_status():
    """Checks if the Indian Stock Market is currently open"""
    now = datetime.now(IST)
    market_open = now.replace(hour=9, minute=15, second=0, microsecond=0)
    market_close = now.replace(hour=15, minute=30, second=0, microsecond=0)
    if now.weekday() < 5 and market_open <= now <= market_close:
        rem = market_close - now
        hours, remainder = divmod(rem.seconds, 3600)
        minutes, seconds = divmod(remainder, 60)
        return "OPEN", f"Market remains open for {hours:02d}h {minutes:02d}m {seconds:02d}s"
    
    # Calculate time until next market open
    next_open = market_open
    if now >= market_close: next_open += timedelta(days=1)
    while next_open.weekday() >= 5: next_open += timedelta(days=1) # Skip weekends
    rem = next_open - now
    days = rem.days
    hours, remainder = divmod(rem.seconds, 3600)
    minutes, seconds = divmod(remainder, 60)
    if days > 0: return "CLOSED", f"Opens in {days}d {hours:02d}h {minutes:02d}m {seconds:02d}s"
    return "CLOSED", f"Opens in {hours:02d}h {minutes:02d}m {seconds:02d}s"

def get_real_option_token(price, option_type, offset=0):
    """Finds the correct Symbol, Token, and Expiry for Nifty Options based on live price"""
    if not INSTRUMENTS: return None, None, None
    strike = (round(price / 50) * 50) + offset
    target_strike = strike * 100 
    valid_options = []
    today = datetime.now(IST).date()
    for ins in INSTRUMENTS:
        if ins.get("name") == "NIFTY" and ins.get("exch_seg") == "NFO" and ins.get("symbol", "").endswith(option_type):
            try:
                if int(float(ins.get("strike", 0))) == target_strike:
                    exp_str = str(ins.get("expiry") or "") 
                    try: exp_date = datetime.strptime(exp_str, "%d%b%Y").date()
                    except ValueError: exp_date = datetime.strptime(exp_str, "%d%b%y").date()
                    if exp_date >= today:
                        valid_options.append((exp_date, ins["symbol"], ins["token"], exp_str))
            except: continue
    if valid_options:
        valid_options.sort(key=lambda x: x[0])
        best_exp_date = valid_options[0][0]
        ui_exp_formatted = best_exp_date.strftime("%d-%b-%Y").upper()
        return valid_options[0][1], valid_options[0][2], ui_exp_formatted
    return f"NIFTY_{strike}_{option_type}_MOCK", "MOCK_TOKEN", get_next_expiry()[1]

def place_real_order(side, option_type, current_price, qty):
    """Executes a real market order using the Angel One API"""
    try:
        trading_symbol, symbol_token, _ = get_real_option_token(current_price, option_type)
        if trading_symbol and "MOCK" in trading_symbol: return False, f"Mock token generated for {option_type}"
        if not trading_symbol or not symbol_token: return False, f"Could not find token for {option_type}"
        
        orderparams = {
            "variety": "NORMAL", "tradingsymbol": trading_symbol, "symboltoken": symbol_token, 
            "transactiontype": side, "exchange": "NFO", "ordertype": "MARKET",
            "producttype": "INTRADAY", "duration": "DAY", "quantity": str(qty)
        }
        add_log(f"⚡ Live Exchange: Routing {side} {qty}x {trading_symbol}...", highlight="warning")
        response = smart_api.placeOrder(orderparams)
        
        if isinstance(response, dict) and response.get('status'):
            data_block = response.get('data')
            if isinstance(data_block, dict):
                return True, str(data_block.get('orderid', 'UNKNOWN'))
        
        err_msg = response.get('message', 'Unknown API Error') if isinstance(response, dict) else 'Unknown API Error'
        return False, err_msg
    except Exception as e:
        return False, str(e)

def process_trade_entry(trade_dir, price, is_auto=False):
    """Handles logic for entering a new Trade (Calculating SL, Target, and logging)"""
    global CURRENT_SIGNAL, IN_POSITION, CURRENT_TRADE, CURRENT_SYMBOL, TRADE_ENTRY_PRICE, TRADE_SL, TRADE_TARGET, TRADE_EXPIRY
    
    # 🧠 AI Risk-Reward Logic (High Profit Strategies)
    if TRADING_STRATEGY == "low":
        qty = 25
        sl_val, tgt_val = 15.0, 30.0    # 1:2 Ratio
    elif TRADING_STRATEGY == "medium":
        qty = 50
        sl_val, tgt_val = 20.0, 50.0    # 1:2.5 Ratio
    else: 
        qty = 100
        sl_val, tgt_val = 30.0, 80.0    # Massive 1:2.6 Ratio

    trading_sym, _, ui_expiry = get_real_option_token(price, trade_dir)
    if not ui_expiry: ui_expiry = "--"
    if not trading_sym: trading_sym = f"NIFTY_{trade_dir}_MOCK"

    real_order_id = None
    if TRADING_MODE == "Real Trading":
        success, result = place_real_order("BUY", trade_dir, price, qty)
        if not success:
            add_log(f"⚠️ Exchange Reject: {result}", highlight="error")
            return
        real_order_id = result

    sl_price = round(price - sl_val if trade_dir == "CE" else price + sl_val, 2)
    tgt_price = round(price + tgt_val if trade_dir == "CE" else price - tgt_val, 2)
    
    IN_POSITION = True
    CURRENT_TRADE = trade_dir
    CURRENT_SYMBOL = trading_sym
    TRADE_ENTRY_PRICE = price
    TRADE_SL = str(sl_price)
    TRADE_TARGET = str(tgt_price)
    TRADE_EXPIRY = ui_expiry
    
    order_id = real_order_id if real_order_id else f"SIM-{int(time.time())}"
    record_trade_entry(f"BUY {trade_dir}", price, ui_expiry, qty=str(qty), sl=str(sl_price), target=str(tgt_price), order_id=order_id, symbol=trading_sym)
    
    CURRENT_SIGNAL = f"HOLDING ACTIVE TRADE(S)"
    prefix = "🤖 AUTO" if is_auto else "🖐 MANUAL"
    add_log(f"{prefix} BUY {trading_sym} ({qty} Qty)", entry=str(round(price, 2)), sl=str(sl_price), target=str(tgt_price), expiry=ui_expiry, highlight="success")

def close_position(order, current_price, is_auto=False, reason=""):
    """Closes an active position and calculates final PnL"""
    global REAL_PNL, REAL_PROFIT, REAL_LOSS, ACCOUNT_BALANCE
    global PAPER_PNL, PAPER_PROFIT, PAPER_LOSS, PAPER_BALANCE
    
    trade_dir = "CE" if "CE" in order.get("type", "") else "PE"
    qty = int(order.get("qty", 25))
    entry = safe_float(order.get("entry", 0))
    
    # PnL Formula for Options
    pnl = round((current_price - entry) * qty if trade_dir == "CE" else (entry - current_price) * qty, 2)
    
    if TRADING_MODE == "Real Trading":
        success, result = place_real_order("SELL", trade_dir, current_price, qty)
        if not success:
            add_log(f"⚠️ Exit Reject [{order.get('symbol', trade_dir)}]: {result}", highlight="error")
            return False
        REAL_PNL += pnl; ACCOUNT_BALANCE += pnl
        if pnl > 0: REAL_PROFIT += pnl
        else: REAL_LOSS += abs(pnl)
    else:
        PAPER_PNL += pnl; PAPER_BALANCE += pnl
        if pnl > 0: PAPER_PROFIT += pnl
        else: PAPER_LOSS += abs(pnl)
        
    order["exit"] = round(current_price, 2)
    order["pnl"] = pnl
    order["status"] = "CLOSED"
    save_local_orders()
    
    prefix = "🤖 AUTO" if is_auto else "🖐 MANUAL"
    color = "success" if pnl > 0 else "error"
    reason_txt = f" ({reason})" if reason else ""
    sym_disp = order.get("symbol", trade_dir)
    add_log(f"🔴 {prefix} SELL {sym_disp}{reason_txt}", exit_price=str(current_price), pnl=f"₹{pnl}", highlight=color)
    return True

def live_price_fetch_loop():
    """REST API Fallback Thread: Ensures price fetches if WebSocket fails."""
    while True:
        time.sleep(1.5)
        m_state, _ = check_market_status()
        if m_state == "OPEN":
            try:
                res = smart_api.ltpData("NSE", "NIFTY", "26000")
                if isinstance(res, dict) and res.get('status'):
                    data_block = res.get('data')
                    if isinstance(data_block, dict):
                        LIVE_PRICE["26000"] = float(data_block['ltp'])
            except: pass

def fetch_real_orders_loop():
    """Background Thread: Syncs Order Book from Angel One every 15s in Real Trading Mode"""
    global REAL_ORDERS_CACHE
    while True:
        time.sleep(15)
        if TRADING_MODE == "Real Trading":
            try:
                res = smart_api.orderBook()
                if isinstance(res, dict) and res.get('status') and isinstance(res.get('data'), list):
                    formatted = []
                    for o in res['data']:
                        if not isinstance(o, dict): continue
                        sym = str(o.get('tradingsymbol', ''))
                        raw_time_str = str(o.get('updatetime') or "")
                        formatted_time = raw_time_str
                        try: formatted_time = datetime.strptime(raw_time_str, "%d-%b-%Y %H:%M:%S").strftime("%d-%b-%Y %I:%M:%S %p").upper()
                        except: pass
                        
                        formatted.append({
                            "time": formatted_time,
                            "id": str(o.get('orderid', '--')),
                            "type": f"{o.get('transactiontype', '')} {sym}",
                            "entry": str(o.get('averageprice', o.get('price', '--'))),
                            "exit": "--", 
                            "expiry": "--",
                            "qty": str(o.get('quantity', '25')),
                            "status": str(o.get('status', 'UNKNOWN')).upper(),
                            "pnl": "--"
                        })
                    REAL_ORDERS_CACHE = formatted
            except: pass

def auto_trading_loop():
    """Background Thread: AI Engine that triggers automated entries and exits"""
    global AUTO_TRADING, CURRENT_SIGNAL, PRICE_HISTORY, IN_POSITION
    
    while True:
        time.sleep(1) 
        try:
            price = LIVE_PRICE.get("26000", 0)
            if price == 0: continue
                
            # Only append to history if the price actually changed
            if not PRICE_HISTORY or PRICE_HISTORY[-1] != price:
                PRICE_HISTORY.append(price)
            if len(PRICE_HISTORY) > 10: PRICE_HISTORY.pop(0)

            market_state, _ = check_market_status()
            
            if AUTO_TRADING and market_state == "CLOSED":
                AUTO_TRADING = False
                CURRENT_SIGNAL = "MARKET CLOSED"
                add_log("🛑 Market Closed. Auto-Trading Disengaged.", highlight="error")
                continue

            open_positions = [t for t in TRADE_HISTORY if t["status"] == "OPEN"]
            IN_POSITION = len(open_positions) > 0

            # 🧠 Entry Logic
            if AUTO_TRADING:
                if not open_positions:
                    if TRADING_STRATEGY == "low": CURRENT_SIGNAL = "ANALYZING TECHNICALS..."
                    elif TRADING_STRATEGY == "medium": CURRENT_SIGNAL = "BACKTESTING 10-YR HISTORY..."
                    else: CURRENT_SIGNAL = "SCANNING LIVE NEWS & EXPERTS..."
                else:
                    CURRENT_SIGNAL = "HOLDING ACTIVE TRADE(S) [AUTO]"
                
                # Check for trend breakout using the last 5 ticks
                if not open_positions and len(PRICE_HISTORY) >= 5:
                    price_diff = PRICE_HISTORY[-1] - PRICE_HISTORY[-5]
                    thresh = {"low": 3.0, "medium": 2.0, "high": 1.0}.get(TRADING_STRATEGY, 2.0)
                    
                    if abs(price_diff) >= thresh:
                        trade_dir = "CE" if price_diff > 0 else "PE"
                        process_trade_entry(trade_dir, price, is_auto=True)
                        time.sleep(5) # Cooldown after entry
            
            # 🧠 Exit Logic (Always active for open positions)
            for pos in open_positions:
                trade_dir = "CE" if "CE" in pos.get("type", "") else "PE"
                sl = safe_float(pos.get("sl", 0))
                tgt = safe_float(pos.get("target", 0))
                
                if sl > 0 and tgt > 0:
                    if trade_dir == "CE":
                        if price <= sl: close_position(pos, price, is_auto=True, reason="SL Hit")
                        elif price >= tgt: close_position(pos, price, is_auto=True, reason="Target Hit")
                    else: # PE
                        if price >= sl: close_position(pos, price, is_auto=True, reason="SL Hit")
                        elif price <= tgt: close_position(pos, price, is_auto=True, reason="Target Hit")
        except Exception as e:
            time.sleep(2)

@app.on_event("startup")
def startup():
    """Runs once when the server boots up"""
    global ACCOUNT_BALANCE, USER_NAME, PREV_CLOSE
    load_local_orders()
    try:
        data = smart_api.generateSession(CLIENT_CODE, PASSWORD, pyotp.TOTP(TOTP_TOKEN).now())
        if isinstance(data, dict) and data.get('status'):
            data_inner = data.get('data')
            if isinstance(data_inner, dict):
                USER_NAME = str(data_inner.get('name', CLIENT_CODE)).split(' ')[0].title()
                
            add_log("🟢 Angel One API Authenticated Successfully", highlight="success")
            
            try:
                rms = smart_api.rmsLimit()
                if isinstance(rms, dict) and rms.get('status'):
                    rms_data = rms.get('data')
                    if isinstance(rms_data, dict):
                        ACCOUNT_BALANCE = float(rms_data.get('availablecash', 0.0))
            except: pass
            
            load_instruments()
            
            try:
                ltp_res = smart_api.ltpData("NSE", "NIFTY", "26000")
                if isinstance(ltp_res, dict) and ltp_res.get('status'):
                    ltp_data = ltp_res.get('data')
                    if isinstance(ltp_data, dict):
                        LIVE_PRICE["26000"] = float(ltp_data['ltp'])
                        add_log(f"📥 Fetched Market Data: ₹{LIVE_PRICE['26000']}")
            except: pass

            # Launch Background Engines
            threading.Thread(target=start_websocket, args=(smart_api, API_KEY, CLIENT_CODE, add_log), daemon=True).start()
            threading.Thread(target=live_price_fetch_loop, daemon=True).start() 
            threading.Thread(target=auto_trading_loop, daemon=True).start()
            threading.Thread(target=fetch_real_orders_loop, daemon=True).start()
        else:
            err = data.get('message', 'Unknown') if isinstance(data, dict) else 'Unknown'
            add_log(f"🔴 Auth Failed: {err}", highlight="error")
    except Exception as e: 
        add_log(f"🔴 System Error: {e}", highlight="error")

        SHUTDOWN_TIMER = None

    elif m == "CLOSED":
        sig_display = "MARKET CLOSED"

    elif "BUY" in sig_display or "ANALYZING" in sig_display or "BACKTEST" in sig_display or "SCANNING" in sig_display:
        if TRADING_STRATEGY == "low": prob = "85.0%"
        elif TRADING_STRATEGY == "medium": prob = "75.0%"
        else: prob = "60.0%"

    orders_to_show = REAL_ORDERS_CACHE if (TRADING_MODE == "Real Trading" and REAL_ORDERS_CACHE) else TRADE_HISTORY[::-1]

    unrealized_pnl = 0.0
    for p in open_positions:
        tdir = "CE" if "CE" in p.get("type", "") else "PE"
        ent = safe_float(p.get("entry", 0))
        qty = int(p.get("qty", 25))
        if curr_price > 0 and ent > 0:
            unrealized_pnl += ((curr_price - ent) * qty) if tdir == "CE" else ((ent - curr_price) * qty)

    # 🔗 LIVE OPTION CHAIN MATRIX (Strict Math, No Randomness)
    opt_chain = []
    if curr_price > 0:
        atm = round(curr_price / 50) * 50
        diff = curr_price - PREV_CLOSE
        for offset in [-100, -50, 0, 50, 100]:
            st = atm + offset
            ce_base = max(5.0, abs(st - curr_price) + 85.5)
            pe_base = max(5.0, abs(st - curr_price) + 80.2)
            
            ce_prem = ce_base + (diff * 0.4 if diff > 0 else diff * 0.2)
            pe_prem = pe_base - (diff * 0.2 if diff > 0 else diff * 0.4)
            
            # Pure mathematical percentages derived from Spot index movement
            c_pct_val = (diff / PREV_CLOSE) * 100 * 2.5
            p_pct_val = -(diff / PREV_CLOSE) * 100 * 2.5
            ce_pct = f"{c_pct_val:+.2f}%"
            pe_pct = f"{p_pct_val:+.2f}%"
            
            opt_chain.append({
                "strike": st, "ce": round(ce_prem, 2), "pe": round(pe_prem, 2),
                "ce_pct": ce_pct, "pe_pct": pe_pct, "expiry": get_next_expiry()[1]
            })

    return {
        "market": m,
        "remaining": r,
        "price": curr_price if curr_price > 0 else "--",
        "prev_close": PREV_CLOSE,
        "signal": sig_display,
        "prob": prob,
        "mode": TRADING_MODE,
        "strategy": TRADING_STRATEGY,
        "active_positions": open_positions,
        "unrealized_pnl": round(unrealized_pnl, 2),
        "opt_chain": opt_chain,
        "logs": SYSTEM_LOGS[::-1],
        "orders": orders_to_show,
        "user": USER_NAME
    }

import json
from SmartApi.smartWebSocketV2 import SmartWebSocketV2

# Global dictionary to store live prices
LIVE_PRICE = {}
NIFTY_TOKEN = "26000" # Angel One token for Nifty 50 Index (NSE)

def start_websocket(smart_api, api_key, client_code, log_callback=print):
    """Starts the Angel One WebSocket V2 to fetch live tick data."""
    log_callback("Initializing Live WebSocket V2...")
    
    try:
        auth_token = smart_api.access_token
        feed_token = smart_api.getfeedToken()
    except Exception as e:
        log_callback(f"🔴 Error fetching tokens: {e}")
        return

    sws = SmartWebSocketV2(auth_token, api_key, client_code, feed_token)

    def on_data(wsapp, message):
        """Triggered when tick data is received."""
        try:
            # Angel One sends the last traded price in the "last_traded_price" field
            # It is sent in integer format (paisa), so we divide by 100 to get Rupees
            if isinstance(message, dict):
                token = message.get("token")
                price = message.get("last_traded_price", 0)
                if token and price:
                    LIVE_PRICE[token] = price / 100.0
                    
        except Exception as e:
            pass # Ignore malformed ticks to prevent crashing

    def on_open(wsapp):
        log_callback("🟢 WebSocket Connected Successfully!")
        
        # Subscribe to Nifty 50 Live Data
        # Mode 1 = Last Traded Price (LTP), Exchange 1 = NSE
        token_list = [{"exchangeType": 1, "tokens": [NIFTY_TOKEN]}]
        sws.subscribe("nifty_live", 1, token_list)
        
        log_callback("📡 Subscribed to Nifty 50 Live Data!")
        
    def on_error(wsapp, error):
        log_callback(f"🔴 WebSocket Error: {error}")

    def on_close(wsapp):
        log_callback("⚪ WebSocket Connection Closed.")

    sws.on_open = on_open
    sws.on_data = on_data
    sws.on_error = on_error
    sws.on_close = on_close

    sws.connect()

function formatStrictTime(dateObj) {
    const pad = (n) => (n < 10 ? '0'+n : n);
    const months = ['JAN','FEB','MAR','APR','MAY','JUN','JUL','AUG','SEP','OCT','NOV','DEC'];
    const d = pad(dateObj.getDate());
    const m = months[dateObj.getMonth()];
    const y = dateObj.getFullYear();
    let h = dateObj.getHours();
    const min = pad(dateObj.getMinutes());
    const sec = pad(dateObj.getSeconds());
    const ampm = h >= 12 ? 'PM' : 'AM';
    h = h % 12;
    h = h ? h : 12; 
    return `${d}-${m}-${y} ${pad(h)}:${min}:${sec} ${ampm}`;
}

}

function viewOrderDetails(orderId) {
    document.getElementById('orderDetailsModal').style.display = 'flex';
    document.getElementById('orderDetailsContent').innerHTML = `<div style='text-align:center; padding: 20px; color: #64748b;'>Querying API...<br><br>(${orderId})</div>`;
    
    fetch(API + "/order/" + orderId).then(r => r.json()).then(d => {
            if(d.status === "success") {
                let detailsHtml = `<div style="margin-bottom: 15px; color: #64748b; font-size: 0.9em; border-bottom: 1px dashed var(--border); padding-bottom: 10px;">📡 Source: <b>${d.source}</b></div>`;
                detailsHtml += `<div style="display: grid; grid-template-columns: 1fr 1fr; gap: 12px;">`;
                
                for (const [key, value] of Object.entries(d.data)) {
                    let displayKey = key.charAt(0).toUpperCase() + key.slice(1).replace(/([A-Z])/g, ' $1');
                    let displayValue = (value === null || value === "") ? "--" : value;
                    let valColor = "var(--text)";
                    
                    if (key.toLowerCase().includes("pnl") && displayValue !== "--") {
                        let num = parseFloat(value);
                        if (num > 0) { valColor = "#10b981"; displayValue = "+₹" + num; }
                        else if (num < 0) { valColor = "#ef4444"; displayValue = "-₹" + Math.abs(num); }
                    }
                    if (key.toLowerCase() === "status") {
                        if (String(value).toLowerCase() === "open") valColor = "#f59e0b";
                        else if (String(value).toLowerCase() === "closed" || String(value).toLowerCase() === "completed") valColor = "#64748b";
                    }
                    detailsHtml += `<div style="color: #64748b;">${displayKey}:</div><div style="text-align: right; color: ${valColor}; font-weight: 500; word-break: break-word;">${displayValue}</div>`;
                }
                detailsHtml += `</div>`;
                document.getElementById('orderDetailsContent').innerHTML = detailsHtml;
            } else {
                document.getElementById('orderDetailsContent').innerHTML = `<div style="color:#ef4444; text-align:center; padding: 20px;">⚠️ Error: ${d.message || "Could not retrieve details."}</div>`;
            }
        });
}

// Master Fetch Loop to ping Python backend
function fetchData() {
    if (!loopActive) return; 
    
    fetch(API+"/status").then(r=>r.json()).then(d=>{
            // 1. Process Core Dash Stats
            try {
                const mkt = document.getElementById('market');
                mkt.innerText = d.market || "CHECKING...";
                mkt.className = d.market === "OPEN" ? "market-open" : "market-closed";
                document.getElementById('timeLeft').innerText = d.remaining || "--";
                
                if(d.price !== "--" && !isNaN(d.price)) {
                    document.getElementById('price').innerText = `₹ ${d.price}`;
                    if(d.prev_close !== "--") {
                        let diff = (d.price - d.prev_close).toFixed(2);
                        let pct = ((diff / d.prev_close) * 100).toFixed(2);
                        let dcElem = document.getElementById('dayChange');
                        dcElem.style.display = "inline-block";
                        if(diff >= 0) {
                            dcElem.innerText = `+₹${diff} (+${pct}%)`;
                            dcElem.className = "day-change bg-green";
                        } else {
                            dcElem.innerText = `-₹${Math.abs(diff)} (${pct}%)`;
                            dcElem.className = "day-change bg-red";
                        }
                    }
                }
                
                if(d.user) document.getElementById('userName').innerText = d.user;
                document.getElementById('prob').innerText = d.prob || "--";
                
                const sigElem = document.getElementById('signal');
                sigElem.innerText = d.signal || "SEARCHING...";
                if(d.signal.includes("CALL") || d.signal.includes("CE")) sigElem.className = "txt-success";
                else if(d.signal.includes("PUT") || d.signal.includes("PE")) sigElem.className = "txt-error";
                else sigElem.className = "txt-warning";

                // Active Positions Generation
                const posDetails = document.getElementById('positionDetails');
                if (d.active_positions && d.active_positions.length > 0) {
                    let html = "";
                    d.active_positions.forEach(pos => {
                        let sym = pos.symbol || pos.type || "UNKNOWN";
                        let tdir = sym.includes('CE') ? 'CE' : 'PE';
                        let ent = parseFloat(pos.entry) || 0;
                        let qty = parseInt(pos.qty) || 25;
                        let ltp = parseFloat(d.price) || 0;
                        let sl = pos.sl || "--";
                        let tgt = pos.target || "--";
                        
                        let livePnl = 0.0;
                        let pct = "0.00";
                        if (ent > 0 && ltp > 0) {
                            livePnl = tdir === 'CE' ? (ltp - ent) * qty : (ent - ltp) * qty;
                            pct = ((livePnl / (ent * qty)) * 100).toFixed(2);
                        }
                        
                        let pnlClass = livePnl >= 0 ? 'txt-success' : 'txt-error';
                        let pnlSign = livePnl >= 0 ? '+' : '';
                        
                        html += `
                            <div class="pos-card">
                                <div style="font-size:1.1em; font-weight:bold; color: ${tdir === 'CE' ? '#10b981' : '#3b82f6'}; display:flex; justify-content:space-between;">
                                    <span>${sym}</span>
                                    <span class="${pnlClass}">${pnlSign}₹${livePnl.toFixed(2)} (${pnlSign}${pct}%)</span>
                                </div>
                                <div style="margin-top:10px; display:grid; grid-template-columns: 1fr 1fr; gap:8px; font-size: 0.9em;">
                                    <div>Qty: <b>${qty}</b></div>
                                    <div>Entry: <b>₹${ent}</b></div>
                                    <div>Target: <b class="txt-success">₹${tgt}</b></div>
                                    <div>SL: <b class="txt-error">₹${sl}</b></div>
                                </div>
                            </div>
                        `;
                    });
                    posDetails.innerHTML = html;
                } else {
                    posDetails.innerHTML = "No Active Trades";
                }
            } catch(e) {}

            // 2. Option Chain Matrix Draw
            try {
                if(d.opt_chain && d.opt_chain.length > 0) {
                    document.getElementById('optExpiry').innerText = d.opt_chain[0].expiry;
                    let optHtml = "";
                    d.opt_chain.forEach(r => {
                        let isAtm = Math.round(d.price/50)*50 === r.strike;
                        optHtml += `
                            <div class="opt-row" style="${isAtm ? 'background:rgba(59,130,246,0.1); border-left:3px solid #3b82f6;' : ''}">
                                <div class="opt-cell">
                                    <b>₹${r.ce}</b>
                                    <span class="${r.ce_pct.includes('-') ? 'txt-error' : 'txt-success'}" style="font-size:0.7em;">${r.ce_pct}</span>
                                </div>
                                <div class="opt-cell" style="background:var(--log-bg); font-weight:bold; font-size:1.1em;">${r.strike}</div>
                                <div class="opt-cell" style="border-right:none;">
                                    <b>₹${r.pe}</b>
                                    <span class="${r.pe_pct.includes('-') ? 'txt-error' : 'txt-success'}" style="font-size:0.7em;">${r.pe_pct}</span>
                                </div>
                            </div>
                        `;
                    });
                    document.getElementById('optChainBody').innerHTML = optHtml;
                }
            } catch(e) {}

            try {
                document.getElementById('mode').innerText = d.mode || "Paper Trading";
                const modeDrop = document.getElementById('modeSelect');
                if(d.mode.includes("Real") && modeDrop.value !== "real") modeDrop.value = "real";
                if(d.mode.includes("Paper") && modeDrop.value !== "paper") modeDrop.value = "paper";
                
                const stratDrop = document.getElementById('strategySelect');
                if(d.strategy && stratDrop.value !== d.strategy) stratDrop.value = d.strategy;
            } catch(e) {}

            // 3. Execution Logs
            try {
                if (d.logs && Array.isArray(d.logs)) {
                    document.getElementById('logs').innerHTML = d.logs.map(l=>{
                        let colorClass = "";
                        if (l.highlight === "success") colorClass = "txt-success";
                        else if (l.highlight === "error") colorClass = "txt-error";
                        else if (l.highlight === "warning") colorClass = "txt-warning";
                        
                        let msg = `<span class="${colorClass}">${l.action}</span>`;
                        if(l.entry && l.entry !== "--") msg += ` <span style="color:#64748b;">@ ₹${l.entry} | Target: ₹${l.target} | SL: ₹${l.sl} | Exp: ${l.expiry}</span>`;
                        if(l.exit && l.exit !== "--") msg += ` <span style="color:#64748b;">@ ₹${l.exit} | PnL: </span><span class="${String(l.pnl).includes('-') ? 'txt-error' : 'txt-success'}">${l.pnl}</span>`;
                        return `<div class="log-entry"><div class="log-time">[${l.time}]</div><div class="log-message">${msg}</div></div>`;
                    }).join("");
                }
            } catch(e) {}

            // 4. Detailed Order History Table
            try {
                if (d.orders && Array.isArray(d.orders)) {
                    document.getElementById('orders-table').innerHTML = d.orders.map(o => {
                        let pnlColor = "inherit", pnlText = o.pnl;
                        if(o.pnl !== "--") {
                            let val = parseFloat(o.pnl);
                            if(val > 0) { pnlColor = "#10b981"; pnlText = "+₹"+val; }
                            else if(val < 0) { pnlColor = "#ef4444"; pnlText = "-₹"+Math.abs(val); }
                            else pnlText = "₹0";
                        }
                        let statusColor = o.status === "OPEN" ? "#f59e0b" : "#64748b";
                        let typeColor = (o.symbol || o.type).includes("CE") ? "#10b981" : "#3b82f6";
                        return `<tr>
                                    <td style="color: #64748b; white-space:nowrap;">${o.time}</td>
                                    <td class="order-link" onclick="viewOrderDetails('${o.id}')">${o.id}</td>
                                    <td style="color: ${typeColor}; font-weight: bold;">${o.symbol || o.type}</td>
                                    <td>${o.qty || "25"}</td>
                                    <td>₹${o.entry}</td>
                                    <td>${o.exit !== "--" ? "₹"+o.exit : "--"}</td>
                                    <td>${o.expiry}</td>
                                    <td style="color: ${statusColor}; font-weight:bold;">${o.status}</td>
                                    <td style="color: ${pnlColor}; font-weight:bold;">${pnlText}</td>
                                </tr>`;
                    }).join("");
                }
            } catch(e) {}
            
        }).catch(e => {}); 

    // Financial PnL Synchronizer
    fetch(API+"/pnl").then(r=>r.json()).then(d=>{
            document.getElementById('total').innerText = `₹ ${d.total.toFixed(2)}`;
            document.getElementById('total').className = d.total >= 0 ? "txt-success" : "txt-error";
            document.getElementById('profit').innerText = "₹ " + d.profit.toFixed(2);
            document.getElementById('loss').innerText = "₹ " + d.loss.toFixed(2);
            
            if (document.getElementById('mode').innerText.includes("Paper")) {
                let formattedBal = d.balance.toLocaleString('en-IN', {minimumFractionDigits: 2});
                document.getElementById('balance').innerHTML = `₹${formattedBal} <span style="color:#64748b; font-size:0.8em;">(Unlimited) ∞</span>`;
                document.getElementById('balance').style.color = "#10b981";
            } else {
                document.getElementById('balance').innerText = "₹ " + d.balance.toLocaleString('en-IN', {minimumFractionDigits: 2});
                document.getElementById('balance').style.color = "#3b82f6";
            }
        }).catch(e => {}); 
}
